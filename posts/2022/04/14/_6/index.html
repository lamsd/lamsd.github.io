# How to build a chatbot with Keras.

# B0. Install the packages.

pip install -r requirements.txt

# B1. Prepare data.

{"intents": [
    {"tag": "xinhchao",
     "patterns": ["Hi", "Hey", "Xin chào!", "Hello", "Hay"],
     "responses": ["Xin chào!", "Hi"]
    },
    {"tag": "tambiet",
     "patterns": ["Tạm biệt!", "Hẹn gặp lại!", "Mình về đây!"],
     "responses": ["Tạm biệt!", "Bửa nào gặp lại!", "Okie! Bửa nào gặp lại!"]
    },
    {"tag": "camon",
     "patterns": ["Cảm ơn!", "Cảm ơn rất nhiều"],
     "responses": ["Cảm ơn rất nhiều!", "Any time!"]
    }
]
}

# B2. Training.

python3 train.py

# 3. Test.

python3 test.py
